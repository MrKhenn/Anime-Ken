function toggleLike(btn) {
  const span = btn.querySelector('span');
  let count = parseInt(span.textContent);
  const dislikeBtn = btn.nextElementSibling;

  if (btn.classList.contains('active-like')) {
    btn.classList.remove('active-like');
    span.textContent = count - 1;
  } else {
    btn.classList.add('active-like');
    span.textContent = count + 1;

    // Quitar dislike si estaba activo
    if (dislikeBtn.classList.contains('active-dislike')) {
      dislikeBtn.classList.remove('active-dislike');
      const dislikeSpan = dislikeBtn.querySelector('span');
      dislikeSpan.textContent = parseInt(dislikeSpan.textContent) - 1;
    }
  }
}

function toggleDislike(btn) {
  const span = btn.querySelector('span');
  let count = parseInt(span.textContent);
  const likeBtn = btn.previousElementSibling;

  if (btn.classList.contains('active-dislike')) {
    btn.classList.remove('active-dislike');
    span.textContent = count - 1;
  } else {
    btn.classList.add('active-dislike');
    span.textContent = count + 1;

    // Quitar like si estaba activo
    if (likeBtn.classList.contains('active-like')) {
      likeBtn.classList.remove('active-like');
      const likeSpan = likeBtn.querySelector('span');
      likeSpan.textContent = parseInt(likeSpan.textContent) - 1;
    }
  }
}
